```python
"""v3 migration: 验证 model 列 + index 的 upgrade / downgrade。"""
from pathlib import Path

import pytest
from sqlalchemy import create_engine, inspect


def _alembic_cmd(args: list[str], db_url: str) -> int:
    """直接调 alembic CLI，避免污染主 session 的 alembic.config 单例。"""
    import subprocess
    env_args = ["uv", "run", "alembic", "-x", f"db_url={db_url}", *args]
    return subprocess.run(env_args, check=False).returncode


@pytest.fixture
def temp_db(tmp_path: Path):
    db_path = tmp_path / "test_v3.db"
    db_url = f"sqlite:///{db_path}"
    # 升到 v3 之前的 head
    rc = _alembic_cmd(["upgrade", "head"], db_url)
    assert rc == 0
    return db_url, db_path


def test_v3_upgrade_adds_model_column_and_index(temp_db):
    db_url, _ = temp_db
    engine = create_engine(db_url)
    cols = {c["name"] for c in inspect(engine).get_columns("assets")}
    assert "model" in cols, "model 列应已加到 assets 表"

    indexes = {i["name"] for i in inspect(engine).get_indexes("assets")}
    assert "ix_assets_model" in indexes, "ix_assets_model 索引应存在"


def test_v3_downgrade_removes_model_column_and_index(temp_db):
    db_url, _ = temp_db
    rc = _alembic_cmd(["downgrade", "-1"], db_url)
    assert rc == 0

    engine = create_engine(db_url)
    cols = {c["name"] for c in inspect(engine).get_columns("assets")}
    assert "model" not in cols, "downgrade 后 model 列应删除"

    indexes = {i["name"] for i in inspect(engine).get_indexes("assets")}
    assert "ix_assets_model" not in indexes
```

```python
def register(
    self,
    name: str,
    type_id: uuid.UUID,
    custom_data: dict,
    serial_number: str | None = None,
    model: str | None = None,  # 新；位置紧邻 sn
    holder: str | None = None,
    location: str | None = None,
    notes: str | None = None,
    acquired_at: date | None = None,
) -> Asset:
    asset_type = self.type_repo.get(type_id)
    if asset_type is None:
        raise NotFoundError(f"类型不存在: {type_id}")

    validated_data = validate_custom_data(asset_type.custom_fields, custom_data)
    asset_code = self._generate_asset_code(asset_type.code_prefix, type_id)

    asset = Asset(
        asset_code=asset_code,
        name=name,
        type_id=type_id,
        serial_number=serial_number,
        model=model,  # 新
        holder=holder,
        location=location,
        notes=notes,
        custom_data=validated_data,
        acquired_at=acquired_at,
    )
    try:
        self.repo.add(asset)
        self.session.commit()
    except IntegrityError as e:
        self.session.rollback()
        orig_msg = (
            str(e.orig.args[0]).lower()
            if e.orig and e.orig.args
            else str(e).lower()
        )
        if "asset_code" in orig_msg:
            raise DuplicateError(f"asset_code 撞车（请重试）: {asset_code}") from None
        raise DuplicateError(f"序列号重复: {serial_number}") from None
    self.session.refresh(asset)
    return asset
```

```python
def update_asset(
    self,
    asset_id: uuid.UUID,
    name: str | None = None,
    serial_number: str | UnsetType = UNSET,
    model: str | None | UnsetType = UNSET,  # 新
    notes: str | UnsetType = UNSET,
    custom_data: dict | UnsetType = UNSET,
    acquired_at: date | None | UnsetType = UNSET,
) -> Asset:
    """更新资产非状态字段。

    M3a 后 status/holder/location 不再走 PATCH——必须通过
    POST /api/assets/{id}/transitions 经 state machine 校验。
    """
    a = self.get_asset(asset_id)
    if name is not None:
        a.name = name
    if not isinstance(serial_number, UnsetType):
        a.serial_number = serial_number
    if not isinstance(model, UnsetType):  # 新
        a.model = model
    if not isinstance(notes, UnsetType):
        a.notes = notes
    if not isinstance(custom_data, UnsetType):
        asset_type = self.type_repo.get(a.type_id)
        a.custom_data = validate_custom_data(asset_type.custom_fields, custom_data)
    if not isinstance(acquired_at, UnsetType):
        a.acquired_at = acquired_at
    a.updated_at = datetime.now(UTC)
    try:
        self.session.commit()
    except IntegrityError:
        self.session.rollback()
        raise DuplicateError("唯一约束冲突") from None
    self.session.refresh(a)
    return a
```
