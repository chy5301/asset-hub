class AssetHubError(Exception):
    """asset-hub 域异常 base class（v2.0 引入结构化字段）。"""

    code: str  # 子类必填

    def __init__(
        self,
        message: str,
        *,
        hint: str | None = None,
        fields_missing: list[str] | None = None,
        fields_invalid: dict[str, str] | None = None,
        affected_resource_id: str | None = None,
    ):
        super().__init__(message)
        self.message = message
        self.hint = hint
        self.fields_missing = fields_missing
        self.fields_invalid = fields_invalid
        self.affected_resource_id = affected_resource_id


class NotFoundError(AssetHubError):
    code = "not_found"


class DuplicateError(AssetHubError):
    code = "duplicate"


class ValidationError(AssetHubError):
    code = "validation"


class StateError(AssetHubError):
    code = "state"


class ConflictError(AssetHubError):
    code = "conflict"


class IllegalTransitionError(AssetHubError):
    code = "illegal_transition"
```

import json
import typer
import click


def print_help_json(ctx: typer.Context) -> None:
    """输出当前 command 的 JSON 结构化 help（agent 友好）。"""
    command: click.Command = ctx.command
    payload = {
        "command": " ".join(ctx.command_path.split()),
        "help": command.help or "",
        "params": [
            {
                "name": opt,
                "type": _type_name(p.type),
                "default": _default_value(p.default),
                "required": p.required,
                "help": p.help or "",
            }
            for p in command.params
            for opt in p.opts
        ],
        "examples": _gather_examples(command),
    }
    print(json.dumps(payload, ensure_ascii=False, indent=2))


def _type_name(t):
    if isinstance(t, click.types.StringParamType):
        return "str"
    if isinstance(t, click.types.IntParamType):
        return "int"
    if isinstance(t, click.types.BoolParamType):
        return "bool"
    return str(t)


def _default_value(d):
    if callable(d):
        return None
    return d


def _gather_examples(command):
    # 从 command.help 或 docstring 末尾提取示例（约定：以 "Examples:" 起的段）
    # 简化实现：返回空 list；plan 阶段需要时再扩

    if fields is None:
        return None
    return {f.strip() for f in fields.split(",") if f.strip()}


def serialize_with_fields(model: BaseModel, fields: set[str] | None, allowed: set[str]) -> dict:
    if fields is None:
        return model.model_dump(mode="json")
    unknown = fields - allowed
    if unknown:
        raise ValidationError(
            f"未知字段: {', '.join(sorted(unknown))}",
            fields_invalid={f: "未知字段" for f in unknown},
            hint=f"合法字段：{', '.join(sorted(allowed))}",
        )
    return model.model_dump(mode="json", include=fields)
```
