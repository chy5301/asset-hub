```python
"""v2 state machine broken reassign

Revision ID: <auto>
Revises: <m3a-baseline>
Create Date: 2026-05-12 ...
"""
from alembic import op
import sqlalchemy as sa


def upgrade() -> None:
    # 1. status enum 加 BROKEN（SQLite CHECK constraint 用 batch）
    with op.batch_alter_table("assets") as batch:
        batch.alter_column(
            "status",
            existing_type=sa.Enum("IDLE", "IN_USE", "MAINTENANCE", "RETIRED", "DISPOSED", name="assetstatus"),
            type_=sa.Enum("IDLE", "IN_USE", "MAINTENANCE", "BROKEN", "RETIRED", "DISPOSED", name="assetstatus"),
            existing_nullable=False,
        )

    # 2. transition kind enum 重写：删 RELOCATE/TRANSFER_HOLDER，加 4 新
    # 先数据迁移（合并 RELOCATE/TRANSFER_HOLDER → REASSIGN）
    op.execute(
        "UPDATE state_transition_records SET kind = 'REASSIGN' "
        "WHERE kind IN ('RELOCATE', 'TRANSFER_HOLDER')"
    )

    # 再改 enum 定义
    with op.batch_alter_table("state_transition_records") as batch:
        batch.alter_column(
            "kind",
            existing_type=sa.Enum(
                "CHECKOUT_INTERNAL", "CHECKOUT_EXTERNAL", "RETURN",
                "SEND_TO_MAINTENANCE", "RECOVER_FROM_MAINTENANCE",
                "RETIRE", "REINSTATE", "DISPOSE",
                "RELOCATE", "TRANSFER_HOLDER",
                name="transitionkind",
            ),
            type_=sa.Enum(
                "CHECKOUT_INTERNAL", "CHECKOUT_EXTERNAL", "RETURN",
                "SEND_TO_MAINTENANCE", "RECOVER_FROM_MAINTENANCE",
                "RETIRE", "REINSTATE", "DISPOSE",
                "REASSIGN",
                "REPORT_BROKEN", "DECLARE_UNREPAIRABLE", "DISMISS",
                name="transitionkind",
            ),
            existing_nullable=False,
        )


def downgrade() -> None:
    # 拒绝 downgrade 如有 BROKEN 状态资产
    conn = op.get_bind()
    result = conn.execute(sa.text("SELECT COUNT(*) FROM assets WHERE status = 'BROKEN'")).scalar()
    if result and result > 0:
        raise RuntimeError(f"拒绝 downgrade：仍有 {result} 个 BROKEN 状态资产，请先迁移其状态")

    # 同样拒绝有 v2.0 新 kind 的 transition record
    result = conn.execute(sa.text(
        "SELECT COUNT(*) FROM state_transition_records "
        "WHERE kind IN ('REASSIGN', 'REPORT_BROKEN', 'DECLARE_UNREPAIRABLE', 'DISMISS')"
    )).scalar()
    if result and result > 0:
        raise RuntimeError(f"拒绝 downgrade：仍有 {result} 个 v2.0 新 kind 的 transition records")

    # 反向 enum（v1.0 集合）
    with op.batch_alter_table("state_transition_records") as batch:
        batch.alter_column(
            "kind",
            type_=sa.Enum(
                "CHECKOUT_INTERNAL", "CHECKOUT_EXTERNAL", "RETURN",
                "SEND_TO_MAINTENANCE", "RECOVER_FROM_MAINTENANCE",
                "RETIRE", "REINSTATE", "DISPOSE",
                "RELOCATE", "TRANSFER_HOLDER",
                name="transitionkind",
            ),
            existing_nullable=False,
        )

    with op.batch_alter_table("assets") as batch:
        batch.alter_column(
            "status",
            type_=sa.Enum("IDLE", "IN_USE", "MAINTENANCE", "RETIRED", "DISPOSED", name="assetstatus"),
            existing_nullable=False,
        )
```

```python
from datetime import UTC, datetime

import pytest
from alembic.config import Config
from alembic import command
from sqlalchemy import create_engine, text

    """造一个 v1.0 baseline 的临时数据库。"""
    db_path = tmp_path / "test.db"

    cfg.set_main_option("script_location", "src/asset_hub/alembic")

    # upgrade 到 v1.0 baseline（上一个 revision）
    command.upgrade(cfg, "<m3a-baseline-rev-id>")  # 实施期填实际 revision id
    return db_path, cfg


def test_upgrade_relocate_to_reassign(v1_db):
    """upgrade 把 RELOCATE / TRANSFER_HOLDER 记录改写为 REASSIGN。"""
    db_path, cfg = v1_db
    engine = create_engine(f"sqlite:///{db_path}")
    asset_id = str(uuid.uuid4())
    now = datetime.now(UTC).isoformat()

    # 插一条 v1.0 RELOCATE 记录
    with engine.begin() as conn:

        ), {"id": asset_id, "tid": str(uuid.uuid4()), "now": now})
        conn.execute(text(
            "INSERT INTO state_transition_records (id, asset_id, kind, from_status, to_status, created_at) "
            "VALUES (:tid, :aid, 'RELOCATE', 'IDLE', 'IDLE', :now)"
        ), {"tid": str(uuid.uuid4()), "aid": asset_id, "now": now})

    # upgrade 到 head
    command.upgrade(cfg, "head")

    # 验证 kind 变 REASSIGN
    with engine.connect() as conn:
        rows = conn.execute(text("SELECT kind FROM state_transition_records")).all()
        assert all(r.kind == "REASSIGN" for r in rows)
```

```python
    _instance = None

    def __new__(cls):
        if cls._instance is None:

    def __repr__(self) -> str:
        return "_UNSET"

    def __bool__(self) -> bool:
```

```tsx
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from "@/components/ui/dialog";
// ...

export function reassignSchema(currentHolder?: string | null, currentLocation?: string | null) {
  return z.object({
    to_holder: z.string().optional(),
    to_location: z.string().optional(),
    note: z.string().optional(),
  }).refine(
    (data) =>
      (data.to_holder !== "" && data.to_holder !== currentHolder) ||
      (data.to_location !== "" && data.to_location !== currentLocation),
    { message: "必须改持有人或位置至少一项" },
  );
}

export function ReassignDialog({ asset, open, onOpenChange }: Props) {
  const form = useForm<z.infer<ReturnType<typeof reassignSchema>>>({
    resolver: zodResolver(reassignSchema(asset.holder, asset.location)),
    defaultValues: { to_holder: asset.holder ?? "", to_location: asset.location ?? "", note: "" },
  });

  const mutation = useTransitionMutation();

  const onSubmit = (data) => {
    mutation.mutate({
      assetId: asset.id,
      body: {
        kind: "REASSIGN",
        ...(data.to_holder !== asset.holder ? { to_holder: data.to_holder || null } : {}),
        ...(data.to_location !== asset.location ? { to_location: data.to_location || null } : {}),
        note: data.note || undefined,
      },
    });
  };

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>重新分配</DialogTitle>
        </DialogHeader>
        <Form {...form}>
          <form onSubmit={form.handleSubmit(onSubmit)}>
            {/* to_holder / to_location / note 三字段，遵循既有 dialog spatial */}
          </form>
        </Form>
      </DialogContent>
    </Dialog>
  );
}
```
